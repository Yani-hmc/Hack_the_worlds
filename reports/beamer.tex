\begin{frame}{Why Self-Supervised EEG?}
\begin{columns}[T]
\column{0.55\textwidth}
\textbf{Clinical reality:}
\begin{itemize}
  \item Every EEG must be reviewed by a licensed neurologist
  \item Labels are scarce and expensive
  \item Unlabelled EEG data is abundant
\end{itemize}

\vspace{0.6em}
\textbf{Goal:} learn a \emph{transferable} EEG representation \textbf{without labels}

\vspace{0.6em}
\textbf{Benchmark — TUAB v3.0.1:}
\begin{itemize}
  \item Binary: normal / abnormal clinical EEG
  \item 19 channels, 200\,Hz, 10\,s windows
  \item 2\,717 train / 276 eval (patient-disjoint)

% ─── 2. CAVERN ALLEGORY (30s) ────────────────────────────────────────────────
\begin{frame}{Plato's Cave — What SSL Actually Does}
\begin{columns}[T]
\column{0.55\textwidth}
Prisoners in Plato's cave see only \textbf{shadows on the wall} — not the real objects casting them.
They learn to \emph{predict and recognise} patterns in the shadows without ever seeing the cause.

\vspace{0.8em}
\textbf{In our setting:}
\begin{itemize}
  \item The cave wall = raw EEG voltages
  \item The hidden reality = underlying brain pathology
  \item The SSL encoder = a prisoner learning that \emph{two corrupted views of the same shadow are the same object}
  \item The linear probe = asking the prisoner to name what they see
\end{itemize}

\vspace{0.5em}
\textcolor{muted}{\footnotesize The encoder never sees a neurologist's label. It learns structure from the signal alone.}

% ─── 3. APPROACH (45s) ───────────────────────────────────────────────────────
\begin{frame}{EB-JEPA — Two-View Invariance + Corruption}
\begin{columns}[T]
\column{0.55\textwidth}
\textbf{Two-view SSL pipeline:}
\begin{enumerate}
  \item Take one EEG window; create two augmented views $v_1, v_2$
  \item Encode both: $z_i = f_\theta(v_i) \in \mathbb{R}^{256}$
  \item Minimise $\|z_1 - z_2\|^2$ + anti-collapse regulariser
\end{enumerate}

\vspace{0.5em}
\textbf{Augmentations (standard):}
Gaussian noise, amplitude jitter, channel drop

\vspace{0.5em}
\textbf{Our design — Corruption augmentation:}
\begin{itemize}
  \item 20\,\% time steps \textbf{masked} to zero
  \item 20\,\% time steps injected with $\pm6\sigma$ \textbf{spikes}
  \item Mimics real EEG artefacts; forces robust representations
\end{itemize}

\vspace{0.5em}
\textbf{Two objectives tested:} VICReg ($V$+$C$ regulariser) and SIGReg (Epps-Pulley Gaussianity test)

\column{0.42\textwidth}
\begin{center}
\begin{tikzpicture}[font=\footnotesize, node distance=0.5cm and 0.7cm]
  \node[draw=gray!50, fill=blue!8, rounded corners, minimum width=3.2cm, align=center]
    (x) {$x$: EEG window};
  \node[draw=blue!50, fill=blue!15, rounded corners, below left=of x, minimum width=2.5cm, align=center]
    (v1) {View 1\\ (masked+spikes)};
  \node[draw=blue!50, fill=blue!15, rounded corners, below right=of x, minimum width=2.5cm, align=center]
    (v2) {View 2\\ (noise+jitter)};
  \node[draw=good!70, fill=good!15, rounded corners, below=0.5cm of v1, minimum width=2.5cm, align=center]
    (z1) {$z_1 \in \mathbb{R}^{256}$};
  \node[draw=good!70, fill=good!15, rounded corners, below=0.5cm of v2, minimum width=2.5cm, align=center]
    (z2) {$z_2 \in \mathbb{R}^{256}$};
  \draw[-{Latex}] (x) -- (v1); \draw[-{Latex}] (x) -- (v2);

% ─── 4. KEY FINDING 1 — CORRUPTION (45s) ────────────────────────────────────
\begin{frame}{Finding 1 — Corruption Augmentation: the Dominant Lever}
\begin{columns}[T]
\column{0.55\textwidth}
\begin{tabular}{lcc}
\toprule
Variant & BACC\textsubscript{REC} & $\Delta$ \\
\midrule
Base VICReg (no corruption) & 0.796 & — \\
\textbf{+ Corruption} (mask + spikes) & \textbf{0.819 ± .004} & \textcolor{good}{+0.023} \\
+ Spectral loss & 0.795--0.813 & $\approx$0 \\
Scaled encoder & 0.805 & $-$0.014 \\
4× more data & 0.812 & $-$0.007 \\
\bottomrule
\end{tabular}

\vspace{0.7em}
\textbf{3-seed average (robust):}\\
$0.819 \pm 0.004$ — not a lucky seed

\column{0.42\textwidth}
\textbf{Why corruption helps:}
\begin{itemize}
  \item Standard augmentation (noise, jitter) creates views that are ``too easy'' to match
  \item Masking + spike injection forces the encoder to learn \emph{content}, not artefact patterns
  \item Mirrors the real EEG noise distribution
\end{itemize}

\vspace{0.5em}
\textbf{What does NOT help:}
\begin{itemize}
  \item More encoder capacity → same plateau
  \item 4× pretraining data → no gain
  \item[$\Rightarrow$] The bottleneck is the \emph{architecture}, not scale
\end{itemize}
\end{columns}
\end{frame}

% ─── 5. KEY FINDING 2 — INV_COEFF (60s) ─────────────────────────────────────
\begin{frame}{Finding 2 (Novel) — EEG Prefers Weak VICReg Invariance}
\begin{columns}[T]
\column{0.55\textwidth}
\textbf{VICReg loss:}
$$\mathcal{L} = \lambda_{\rm inv}\underbrace{\|z_1{-}z_2\|^2}_{\rm invariance}
             + 25\,V + 1\,C$$

Paper-correct: $\lambda_{\rm inv} = 25$. \quad Framework default: $\lambda_{\rm inv} = 1$.

\vspace{0.7em}
\begin{tabular}{lcc}
\toprule
$\lambda_{\rm inv}$ & BACC\textsubscript{REC} & Seeds \\
\midrule
$\lambda = 1$ (default) & \textbf{0.819 ± .004} & 3 \\
$\lambda = 25$ (paper) & 0.789 ± .005 & 3 \\
\bottomrule
\multicolumn{3}{l}{\footnotesize Gap: 3\,pp, ${\approx}6\sigma$ — not noise}
\end{tabular}

\column{0.42\textwidth}
\textbf{Why?}
\begin{itemize}
  \item With $\lambda{=}1$, anti-collapse ($V$+$C$) \textbf{dominates} invariance
  \item Encoder focuses on \emph{spreading} representations across dimensions — more discriminative
  \item With $\lambda{=}25$, invariance dominates: encoder collapses toward view-similarity at the cost of feature diversity
\end{itemize}

\vspace{0.5em}
\textbf{Ongoing ablation:} $\lambda \in \{1, 5, 10, 25\}$\\
(capDinv5 / capDinv10 running now)

\vspace{0.5em}
\textcolor{accent}{This is, to our knowledge, the first systematic evidence that EEG SSL requires weaker view-invariance than vision SSL.}
\end{columns}
\end{frame}

% ─── 6. NAIVEJEPA (45s) ──────────────────────────────────────────────────────
\begin{frame}{NaiveJEPA — Temporal-Split Prediction, No Anti-Collapse Needed}
\begin{columns}[T]
\column{0.55\textwidth}
\textbf{Idea}: split one window in time instead of augmenting.

\vspace{0.4em}
\begin{itemize}
  \item Context ($0\to T/2$): online encoder $\to$ mean-pool
  \item Target ($T/2\to T$): EMA encoder $\to$ mean-pool
  \item Loss: $\mathrm{MSE}(\hat{z}_{\rm tgt},\ z_{\rm tgt})$
  \item \textbf{No VICReg, no SIGReg} needed: context and target see \emph{disjoint time spans} — collapse to a constant is architecturally impossible
\end{itemize}

\vspace{0.5em}
\begin{tabular}{lcc}
\toprule
Method & BACC\textsubscript{REC} & Hyperparams \\
\midrule
VICReg + corruption & 0.819 ± .004 & 3 loss weights + EMA \\
\textbf{NaiveJEPA} & \textbf{0.779} & \textbf{1 (EMA only)} \\
\bottomrule
\end{tabular}

\column{0.42\textwidth}
\begin{center}

% ─── 7. ARCHITECTURE COMPARISON (30s) ────────────────────────────────────────
\begin{frame}{Encoder Ranking — Simple Conv Wins}
\begin{columns}[T]
\column{0.52\textwidth}
\begin{tabular}{llcc}
\toprule
Encoder & Type & Params & BACC\textsubscript{REC} \\
\midrule
\textcolor{good}{\textbf{Conv1D}} & Strided conv & 0.4M & \textbf{0.819} \\
BIOT-style & FFT patches & 1.2M & 0.790 \\
LaBraM-style & Raw patches & 1.2M & $\sim$0.775$^\dagger$ \\
\textcolor{warn}{EEGPT} & Ch-pool Tr & 0.8M & \textcolor{warn}{collapsed} \\
\bottomrule
\multicolumn{4}{l}{\footnotesize $^\dagger$ accuracy, not BACC (early eval)}
\end{tabular}

\vspace{0.5em}
\textbf{Why Conv wins:}\\
The 0.4M conv has no architectural mismatch with corruption augmentation, trains stably, and plateaus at 0.82 regardless of depth, width, or data volume.

\column{0.45\textwidth}
\textbf{EEGPT failure: architectural conflict}
\begin{itemize}
  \item Cross-attention pools 19 channels $\to$ 1 token per time step
  \item Channel-drop creates views with \emph{different channel sets}
  \item The two views are structurally incompatible before the encoder sees them
  \item Fix: remove channel-drop $\to$ TUEV 0.182 $\to$ 0.358
\end{itemize}

\vspace{0.3em}
\textcolor{muted}{\footnotesize Lesson: always ablate augmentations \emph{per} encoder family.}
\end{columns}
\end{frame}

% ─── 8. FAIR COMPARISON TO LITERATURE (45s) ──────────────────────────────────
\begin{frame}{Honest Position vs Literature (Per-Window)}

\begin{alertblock}{The evaluation trap: per-window vs per-recording ($\approx$5\,pp gap)}
Literature (BIOT, LaBraM, FEMBA) reports \textbf{per-window} BACC.
Our evaluation mean-pools 16 windows/patient $\to$ \textbf{per-recording}, which is 5\,pp higher for the same encoder. Comparing them manufactures a false win.
\end{alertblock}

\vspace{0.5em}
\begin{columns}[T]

\textbf{Honest reading:}
\begin{itemize}

  \item \textcolor{warn}{$\times$} 3.9\,pp below LaBraM-Base (5.8M, 200 ep)
  \item Competitive with EEGNet (supervised) as a \emph{frozen} representation

\vspace{0.5em}
\textcolor{accent}{Transfer from a binary task to a 6-class task — with only a logistic regression — is the strongest evidence that the SSL representation is genuinely general.}
\end{columns}

% ─── 10. SUMMARY (30s) ───────────────────────────────────────────────────────
\begin{frame}{Summary}

\begin{columns}[T]
\column{0.50\textwidth}
\textbf{\textcolor{good}{What works}}
\begin{itemize}
  \item \textbf{Corruption augmentation}: +2.3\,pp BACC (3-seed, robust)
  \item \textbf{$\lambda_{\rm inv}{=}1$}: beats paper-correct $\lambda{=}25$ by 3\,pp/$6\sigma$ — EEG-specific finding
  \item \textbf{SIGReg $\geq$ VICReg}: consistent +0.5--1\,pp AUROC
  \item \textbf{NaiveJEPA}: 0.779 BACC with 1 hyperparameter (no anti-collapse term)
  \item \textbf{TUEV transfer}: beats two supervised baselines without TUEV labels
\end{itemize}

\column{0.48\textwidth}
\textbf{\textcolor{warn}{What doesn't}}
\begin{itemize}
  \item Scaling the encoder (capacity is not the limit)
  \item 4$\times$ pretraining data (conv saturates)
  \item EEGPT (channel-drop $\times$ channel-pool conflict)
  \item Spectral auxiliary loss (neutral, not beneficial)
  \item GRU predictive JEPA (same-window prediction degenerates)
\end{itemize}
\end{columns}

\vspace{0.5em}
\begin{block}{Open next step}
$\lambda_{\rm inv}$ sweep $\{1,5,10,25\}$ running now. If monotone ordering confirms, EEG SSL prefers weaker invariance — first systematic evidence in the literature.
\end{block}
\end{frame}
